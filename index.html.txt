    <script>
        let selectedText = "";

        document.addEventListener("DOMContentLoaded", () => {
            if (localStorage.getItem("letters_theme") === "dark") {
                document.documentElement.setAttribute("data-theme", "dark");
            }

            // Matn belgilanishini kuzatish
            document.addEventListener("selectionchange", () => {
                const selection = window.getSelection();
                const text = selection.toString().trim();
                
                if (text.length > 0) {
                    selectedText = text;
                    document.getElementById("speak-menu").style.display = "flex";
                } else {
                    document.getElementById("speak-menu").style.display = "none";
                }
            });
        });

        // GOOGLE TTS ORQALI XITOYCHA O'QISH
        function speakSelectedText() {
            if (!selectedText) return;

            if ('speechSynthesis' in window) {
                window.speechSynthesis.cancel();
                const utterance = new SpeechSynthesisUtterance(selectedText);
                utterance.lang = 'zh-CN'; // Xitoycha Google talaffuzi
                utterance.rate = 0.8;    // Toza va aniq eshitilishi uchun sekinroq tezlik
                
                window.speechSynthesis.speak(utterance);
            }
            
            // O'qib bo'lingach menyuni berkitish
            document.getElementById("speak-menu").style.display = "none";
            window.getSelection().removeAllRanges(); // Belgilashni yechib yuborish
        }

        // MAVZUNI O'ZGARTIRISH (OQ / QORA)
        function toggleTheme() {
            const currentTheme = document.documentElement.getAttribute("data-theme");
            if (currentTheme === "dark") {
                document.documentElement.removeAttribute("data-theme");
                localStorage.setItem("letters_theme", "light");
            } else {
                document.documentElement.setAttribute("data-theme", "dark");
                localStorage.setItem("letters_theme", "dark");
            }
        }
    </script>
